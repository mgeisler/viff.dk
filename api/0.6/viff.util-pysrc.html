 97      """Lift a function to handle deferred arguments. 
 98   
 99      Use this as a decorator. The decorated function accepts the same 
100      arguments as the original function, but arguments for the lifted 
101      function can be Deferreds. The return value of the lifted function 
102      will always be a Deferred. 
103   
104      Keyword arguments are not lifted. 
105   
106      As an example, here is how to define a lazy addition and 
107      multiplication which works for integers (deferred or not): 
108   
109      >>> @dlift 
110      ... def add(a, b): 
111      ...     return a + b 
112      ... 
113      >>> @dlift 
114      ... def mul(a, b): 
115      ...     return a * b 
116      ... 
117      >>> x = Deferred() 
118      >>> y = Deferred() 
119      >>> z = mul(add(x, 10), y) 
120      >>> x.callback(5) 
121      >>> y.callback(10) 
122      >>> z                                         # doctest: +ELLIPSIS 
123      <DeferredList at 0x...  current result: 150> 
124      """ 
125      def lifted(*args, **kwargs): 
126          """Lifted wrapper function.""" 
127          deferred_args = [] 
128          for arg in args: 
129              if not isinstance(arg, Deferred): 
130                  arg = succeed(arg) 
131              deferred_args.append(arg) 
132   
133          # One might opt to lift any keyword arguments too, but it has 
134          # been left out for now since it is somewhat complicated with 
135          # multiple DeferredLists waiting on each other. 
136   
137          results = gatherResults(deferred_args) 
138          results.addCallback(lambda results: func(*results, **kwargs)) 
139          return results
